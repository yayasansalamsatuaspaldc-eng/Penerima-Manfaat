// ====== FUNGSI PENYALURAN (Diperbarui untuk Upload File) ====== function lengkapiBuktiPenyaluran(data) { try { var sheet = getSheet_('Penyaluran'); var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var idCol = headers.indexOf('ID'); var realisasiCol = headers.indexOf('TanggalRealisasi'); var statusCol = headers.indexOf('Status'); var buktiCol = headers.indexOf('BuktiFotoURL'); var kwitansiCol = headers.indexOf('KwitansiURL'); // ID Folder Drive tujuan untuk foto, kwitansi, dan PDF var folderId = '1kEFXyyEVycJopx2CtitUPfskzI77hjJV'; var folder = DriveApp.getFolderById(folderId); var fotoUrl = ''; var kwitansiUrl = ''; // Proses konversi dan unggah Foto ke Google Drive if (data.fotoBase64) { var fotoBlob = Utilities.newBlob(Utilities.base64Decode(data.fotoBase64), data.fotoMime, 'Foto_' + data.penyaluranId + '_' + data.fotoName); var fotoFile = folder.createFile(fotoBlob); fotoFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); fotoUrl = fotoFile.getUrl(); } // Proses konversi dan unggah Kwitansi ke Google Drive if (data.kwitansiBase64) { var kwitBlob = Utilities.newBlob(Utilities.base64Decode(data.kwitansiBase64), data.kwitansiMime, 'Kwitansi_' + data.penyaluranId + '_' + data.kwitansiName); var kwitFile = folder.createFile(kwitBlob); kwitFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); kwitansiUrl = kwitFile.getUrl(); } // Update data link di Spreadsheet for (var i = 1; i < rows.length; i++) { if (rows[i][idCol] === data.penyaluranId) { sheet.getRange(i + 1, realisasiCol + 1).setValue(data.tanggalRealisasi || new Date()); sheet.getRange(i + 1, statusCol + 1).setValue('Selesai'); if (fotoUrl) sheet.getRange(i + 1, buktiCol + 1).setValue(fotoUrl); if (kwitansiUrl) sheet.getRange(i + 1, kwitansiCol + 1).setValue(kwitansiUrl); return true; } } throw new Error('Data penyaluran tidak ditemukan'); } catch (e) { throw new Error('Gagal memproses unggahan bukti: ' + e.message); } } // ====== GENERATE LAPORAN PDF (Diperbarui dengan Folder ID & Lampiran) ====== function generateLaporanPDF(programId) { var detail = getProgramDetail(programId); var p = detail.program; if (!p) throw new Error('Program tidak ditemukan'); // Menggunakan Folder ID yang sudah ditentukan var folderId = '1kEFXyyEVycJopx2CtitUPfskzI77hjJV'; var folder = DriveApp.getFolderById(folderId); var doc = DocumentApp.create('LPJ - ' + p.Nama + ' - ' + Utilities.formatDate(new Date(), 'GMT+7', 'dd-MM-yyyy')); var body = doc.getBody(); body.appendParagraph('LAPORAN PERTANGGUNGJAWABAN') .setHeading(DocumentApp.ParagraphHeading.TITLE) .setAlignment(DocumentApp.HorizontalAlignment.CENTER); body.appendParagraph(p.Nama) .setHeading(DocumentApp.ParagraphHeading.HEADING1) .setAlignment(DocumentApp.HorizontalAlignment.CENTER); body.appendParagraph(' '); body.appendParagraph('Deskripsi Program: ' + (p.Deskripsi || '-')); body.appendParagraph('Periode: ' + formatTgl_(p.TanggalMulai) + ' s/d ' + formatTgl_(p.TanggalSelesai)); body.appendParagraph('Target Dana: Rp ' + formatRp_(p.TargetDana)); body.appendParagraph(' '); body.appendParagraph('RINCIAN DONASI MASUK').setHeading(DocumentApp.ParagraphHeading.HEADING2); var tabelDonasi = [['No', 'Nama Donatur', 'Jumlah', 'Tanggal', 'Metode']]; detail.donasi.forEach(function (d, i) { tabelDonasi.push([String(i + 1), d.NamaDonatur, 'Rp ' + formatRp_(d.Jumlah), formatTgl_(d.Tanggal), d.Metode || '-']); }); if (detail.donasi.length) body.appendTable(tabelDonasi); body.appendParagraph('Total Donasi Masuk: Rp ' + formatRp_(detail.totalMasuk)).editAsText().setBold(true); body.appendParagraph(' '); body.appendParagraph('RINCIAN PENYALURAN BANTUAN').setHeading(DocumentApp.ParagraphHeading.HEADING2); var tabelPenyaluran = [['No', 'Penerima Manfaat', 'Wilayah', 'PIC', 'Bantuan', 'Nilai', 'Tgl Realisasi', 'Status', 'Bukti']]; detail.penyaluran.forEach(function (py, i) { var bukti = (py.BuktiFotoURL ? 'Foto ✓' : 'Foto -') + ' / ' + (py.KwitansiURL ? 'Kwitansi ✓' : 'Kwitansi -'); tabelPenyaluran.push([ String(i + 1), py.NamaPenerima, py.Wilayah || '-', py.PIC || '-', py.ItemBantuan, 'Rp ' + formatRp_(py.NilaiRp), py.Status === 'Selesai' ? formatTgl_(py.TanggalRealisasi) : '-', py.Status || 'Terjadwal', bukti ]); }); if (detail.penyaluran.length) body.appendTable(tabelPenyaluran); body.appendParagraph('Total Sudah Tersalurkan : Rp ' + formatRp_(detail.totalTersalurkan)).editAsText().setBold(true); body.appendParagraph('Total Masih Terjadwal : Rp ' + formatRp_(detail.totalTerjadwal)); body.appendParagraph(' '); body.appendParagraph('RINGKASAN').setHeading(DocumentApp.ParagraphHeading.HEADING2); body.appendParagraph('Total Dana Masuk : Rp ' + formatRp_(detail.totalMasuk)); body.appendParagraph('Total Sudah Tersalurkan : Rp ' + formatRp_(detail.totalTersalurkan)); body.appendParagraph('Total Masih Terjadwal : Rp ' + formatRp_(detail.totalTerjadwal)); body.appendParagraph('Sisa Saldo Belum Terpakai: Rp ' + formatRp_(detail.saldo)).editAsText().setBold(true); body.appendParagraph(' '); // Lampiran Foto & Kwitansi var selesaiPenyaluran = detail.penyaluran.filter(function(py) { return py.Status === 'Selesai' && (py.BuktiFotoURL || py.KwitansiURL); }); if (selesaiPenyaluran.length > 0) { body.appendPageBreak(); body.appendParagraph('LAMPIRAN BUKTI PENYALURAN') .setHeading(DocumentApp.ParagraphHeading.HEADING1) .setAlignment(DocumentApp.HorizontalAlignment.CENTER); body.appendParagraph(' '); selesaiPenyaluran.forEach(function(py, i) { body.appendParagraph(String(i + 1) + '. Penerima: ' + py.NamaPenerima + ' (' + (py.Wilayah || '-') + ') - ' + py.ItemBantuan).setBold(true); var table = body.appendTable(); var tr = table.appendTableRow(); var tdFoto = tr.appendTableCell().setWidth(250); var tdKwitansi = tr.appendTableCell().setWidth(250); table.setBorderWidth(0); tdFoto.appendParagraph('Bukti Foto:'); if (py.BuktiFotoURL) { var blobFoto = getBlobFromUrl_(py.BuktiFotoURL); if (blobFoto) { try { var imgF = tdFoto.appendImage(blobFoto); var maxW = 200; if (imgF.getWidth() > maxW) { imgF.setHeight(imgF.getHeight() * (maxW / imgF.getWidth())); imgF.setWidth(maxW); } } catch(e) { tdFoto.appendParagraph('(Gagal memuat format gambar)'); } } else { tdFoto.appendParagraph('(Link gambar tidak valid)'); } } else { tdFoto.appendParagraph('-'); } tdKwitansi.appendParagraph('Kwitansi:'); if (py.KwitansiURL) { var blobKwit = getBlobFromUrl_(py.KwitansiURL); if (blobKwit) { try { var imgK = tdKwitansi.appendImage(blobKwit); var maxW = 200; if (imgK.getWidth() > maxW) { imgK.setHeight(imgK.getHeight() * (maxW / imgK.getWidth())); imgK.setWidth(maxW); } } catch(e) { tdKwitansi.appendParagraph('(Gagal memuat format gambar)'); } } else { tdKwitansi.appendParagraph('(Link gambar tidak valid)'); } } else { tdKwitansi.appendParagraph('-'); } body.appendParagraph(' '); }); } body.appendParagraph('Dicetak pada: ' + Utilities.formatDate(new Date(), 'GMT+7', 'dd MMMM yyyy HH:mm')).setItalic(true); doc.saveAndClose(); var pdfBlob = DriveApp.getFileById(doc.getId()).getAs('application/pdf'); var pdfFile = folder.createFile(pdfBlob); pdfFile.setName('LPJ - ' + p.Nama + '.pdf'); DriveApp.getFileById(doc.getId()).setTrashed(true); return pdfFile.getUrl(); }
